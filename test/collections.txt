# Tuples

()
# There's no (,)

(a) # NOT a tuple
(a,)
(a,b,c)
(a,b,c,)


==>
Program(
  TupleExpression,
  LineComment,
  ParenExpression(Identifier),
  LineComment,
  TupleExpression(Identifier),
  TupleExpression(Identifier, Identifier, Identifier)
  TupleExpression(Identifier, Identifier, Identifier)
)


# Named tuples

(a = 1,)
(a = 1) # NOT a tuple
(a = 1, b = 2)

==>
Program(
  TupleExpression(KwArg(Identifier, AssignmentOp, IntegerLiteral)),
  ParenExpression(Assignment(Identifier, AssignmentOp, IntegerLiteral)),
  LineComment,
  TupleExpression(
    KwArg(Identifier, AssignmentOp, IntegerLiteral),
    KwArg(Identifier, AssignmentOp, IntegerLiteral),
  )
)


# Named tuples with leading semicolon

(;)
(; a)
(; a, b)
(; a = 1)
(; a = 1, b = 2)
(; a, foo.b)

==>
Program(
  TupleExpression(KeywordArguments),
  TupleExpression(KeywordArguments(Identifier)),
  TupleExpression(KeywordArguments(Identifier, Identifier)),
  TupleExpression(KeywordArguments(KwArg(Identifier, AssignmentOp, IntegerLiteral))),
  TupleExpression(KeywordArguments(KwArg(Identifier, AssignmentOp, IntegerLiteral), KwArg(Identifier, AssignmentOp, IntegerLiteral))),
  TupleExpression(KeywordArguments(Identifier, FieldExpression(Identifier, Field(Identifier)))),
)

# Comprehension arrays

[x for x in xs]
UInt[b(c, e) for c in d for e in f]

==>
Program(
  ComprehensionExpression(
    Generator(
      Identifier,
      GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier)),
    )
  ),
  IndexExpression(
    Identifier,
    ComprehensionExpression(
      Generator(
        CallExpression(Identifier, Arguments(Identifier, Identifier)),
        GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier)),
        GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier)),
      )
    )
  )
)


# Generator expressions

(b(c, e) for c in d if e)
(b(c, e) for c in d, e = 5)
(b(c, e) for c in d, e = 5 if e)

==>
Program(
  ParenExpression(
    Generator(
      CallExpression(Identifier, Arguments(Identifier, Identifier)),
      GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier)),
      GenFilter(if, Identifier),
    )
  ),
  ParenExpression(
    Generator(
      CallExpression(Identifier, Arguments(Identifier, Identifier)),
      GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier), ForBinding(Identifier, AssignmentOp, IntegerLiteral)),
    )
  ),
  ParenExpression(
    Generator(
      CallExpression(Identifier, Arguments(Identifier, Identifier)),
      GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier), ForBinding(Identifier, AssignmentOp, IntegerLiteral)),
      GenFilter(if, Identifier),
    )
  ),
)


# Generator expressions in function calls

f(i for i in iter)
f(1, 2, x for x = xs)
f(1, 2, x + y for x = xs, y = ys)

==>
Program(
  CallExpression(
    Identifier,
    Arguments(
      Generator(
        Identifier,
        GenFor(for, ForBinding(Identifier, AssignmentOp(in), Identifier)),
      )
    )
  )
  CallExpression(
    Identifier,
    Arguments(
      IntegerLiteral,
      IntegerLiteral,
      Generator(
        Identifier,
        GenFor(for, ForBinding(Identifier, AssignmentOp, Identifier)),
      )
    )
  ),
  CallExpression(
    Identifier,
    Arguments(
      IntegerLiteral,
      IntegerLiteral,
      Generator(
        BinaryExpression(
          Identifier,
          Identifier,
        )
        GenFor(for, ForBinding(Identifier, AssignmentOp, Identifier), ForBinding(Identifier, AssignmentOp, Identifier)),
      )
    )
  )
)
